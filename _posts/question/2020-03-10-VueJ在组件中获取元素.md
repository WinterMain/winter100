<pre><code>Vue.component('somecomponent', {<font></font>
    template: '#somecomponent',<font></font>
    props: [...],<font></font>
<font></font>
   ...<font></font>
<font></font>
    created: function() {<font></font>
        somevariablehere.querySelector('input').focus();<font></font>
    }<font></font>
});<font></font>
</code></pre>
