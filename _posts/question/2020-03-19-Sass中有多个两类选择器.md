<pre><code>$pages: shop, contact, about, faq;<font></font>
<font></font>

  @each $page in $pages {<font></font>
    &amp;.#{$page} {<font></font>
      background-color:#FFF;<font></font>
    }<font></font>
  }<font></font>
}<font></font>
</code></pre>
