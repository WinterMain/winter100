<pre class="lang-js prettyprint-override"><code>module.exports = {<font></font>

    "react" : "react"<font></font>
},<font></font>
plugins: [<font></font>
    new webpack.optimize.CommonsChunkPlugin("react", "react.bundle.js")<font></font>
   ]<font></font>
}<font></font>
</code></pre></div>
