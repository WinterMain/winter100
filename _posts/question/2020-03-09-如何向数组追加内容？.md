<pre class="lang-js prettyprint prettyprinted" style=""><code><span class="kwd">if</span><span class="pun">(!</span><span class="pln">myArray</span><span class="pun">[</span><span class="pln">i</span><span class="pun">]){</span><span class="pln"> </span><span class="kwd">continue</span><span class="pun">;</span><span class="pln"> </span><span class="pun">}</span></code></pre>
