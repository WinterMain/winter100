<pre><code>var ip = req.connection.remoteAddress;<font></font>
<font></font>
if (ip.length &lt; 15) <font></font>
{   <font></font>
   ip = ip;<font></font>
}<font></font>
else<font></font>

   var nyIP = ip.slice(7);<font></font>
