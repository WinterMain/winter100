          <div class="discuss-comment"><pre><code>function printdiv(printdivname) {<font></font>
    var headstr = "&lt;html&gt;&lt;head&gt;&lt;title&gt;Booking Details&lt;/title&gt;&lt;/head&gt;&lt;body&gt;";<font></font>
