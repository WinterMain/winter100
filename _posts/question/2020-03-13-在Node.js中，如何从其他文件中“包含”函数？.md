    //data.say();<font></font>
    //console.log(data.toString());<font></font>
}<font></font>
</code></pre>

<p>In <code>fetchingfile.js</code> write this code:</p>

<pre><code>function myerror(){<font></font>

}<font></font>
<font></font>
