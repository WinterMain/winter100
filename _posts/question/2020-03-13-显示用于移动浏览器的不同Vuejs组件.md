<pre><code>methods: {<font></font>
 isMobile() {<font></font>
   if(/Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent)) {<font></font>
     return true<font></font>
   } else {<font></font>
     return false<font></font>
   }<font></font>
 }<font></font>
}<font></font>
