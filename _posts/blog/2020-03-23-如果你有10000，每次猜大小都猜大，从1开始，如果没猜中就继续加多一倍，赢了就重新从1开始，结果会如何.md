  <h2>结论</h2>
  <p>理论上产生大小的概率是相同的，这一次产生大也不会影响下一次产生的结果，我们可以看到程序运行到最后，资金无一例外就会耗尽。为什么会产生这样的结果，主要有以下几个原因：</p>
  <ul>
    <li>1.资金是有限的</li>
    <li>2.最小的2对数到14倍开始就已经是非常大的量了</li>
    <li>3.概率是平等的，每一次都不会互相影响</li>
  </ul>
  <p>所以赌徒心里是非常不可取的，这就像黑洞一样，到最后终将走向失败。这可以映射到生活中很多案例，人还是应当务实，实事求是，脚踏实地，好好学习天天向上。欲戴王冠必承其重，没有什么是可以轻而易举不经努力的。</p>

  <script>
    var total = 1000;
    var count = 0;
    var basicBet = 1;
    var multiple = 2;
    var pause = false;
    var totalArr = [];
    var countArr = [];
    var betArr = [];
    var resultArr = [];
    var myChartTotal = echarts.init(document.getElementById('main'));
    var minTotal = total;
    var maxTotal = total;
    var btnStart = document.getElementById("btnStart");
    var labelResult = document.getElementById("labelResult");
    btnStart.onclick = function() {
      if(pause) {
        btnStart.innerText = "暂停";
        if(betArr[betArr.length - 1] > totalArr[totalArr.length - 1]) {
          start();
        } else {
          pause = false;
          guess(betArr[betArr.length - 1])
        }
      } else {
        btnStart.innerText = "开始";
        pause = true;
      }
    };
    document.getElementById("btnRestart").onclick = function() {
      start();
    };

    function start() {
      count = 0;
      total = parseFloat(document.getElementById("inputInitial").value);
      basicBet = parseFloat(document.getElementById("inputTou").value);
      multiple = parseFloat(document.getElementById("inputMultiple").value);
      minTotal = total;
      maxTotal = total;
      pause = true;
      totalArr = [total];
      resultArr = [];
      countArr = [0];
      betArr = [1];

      setTimeout(function() {
        pause = false;
        guess(basicBet);
      }, 200)
    }
    function guess(bet) {
      var openNum = Math.random();
      count ++;
      countArr.push(count);
      betArr.push(bet);
      if(openNum > 0.5) {
        resultArr.push("大");
        labelResult.innerText = (count + "次：" + "大");
        total += bet;
        totalArr.push(total);
        if(minTotal > total) {
          minTotal = total;
        }
        if(maxTotal < total) {
          maxTotal = total;
        }
        updateChartTotal();
        if(pause) {
          return;
        }
        setTimeout(function() {
          guess(basicBet);
        }, 100)
      } else {
        total = total - bet;
        resultArr.push("小");
        labelResult.innerText = (count + "次：" + "小");
        if(minTotal > total) {
          minTotal = total;
        }
        if(maxTotal < total) {
          maxTotal = total;
        }
        totalArr.push(total);
        updateChartTotal();
        bet = multiple * bet;
        if(bet > total) {
          // Game Over
        } else {
          if(pause) {
            return;
          }
          setTimeout(function() {
            guess(bet);
          }, 100)
        }
      }
    }

    function updateChartTotal() {
      var option = {
        tooltip: {
          trigger: 'axis',
          axisPointer: {
              type: 'cross',
              crossStyle: {
                  color: '#999'
              }
          },
        },
        legend: {
          data: ['总资金', '投注']
        },
        xAxis: {
            type: 'category',
            data: countArr
        },
        yAxis: [
          {
            type: 'value',
            name: '总资金',
          },
          {
            type: 'value',
            name: '投注',
          }
        ],
        series: [
          {
            data: totalArr,
            type: 'line',
            min: minTotal,
            max: maxTotal,
            smooth: true,
            name: "总资金",
            itemStyle: {
              color: "#80a7ea",
            },
            areaStyle: {
              color: new echarts.graphic.LinearGradient(0, 0, 0, 1, [{
                offset: 0,
                color: "#cfdef9",
              }, {
                offset: 1,
                color: "#fff",
              }]),
            },
          },
          {
            data: betArr,
            type: 'line',
            smooth: true,
            yAxisIndex: 1,
            name: "投注",
            itemStyle: {
              color: "rgba(255, 0, 0, 0.5)"
            }
          },
        ]
      };

      // 使用刚指定的配置项和数据显示图表。
      myChartTotal.setOption(option);
    }

    start();
